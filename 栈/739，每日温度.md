var dailyTemperatures = function(temperatures) {
    const length = temperatures.length;
    let res = new Array(length).fill(0)
    let stack = [];
    for(i=0;i<length;i++){
        while(stack.length && temperatures[stack[stack.length-1]] < temperatures[i]){
            const j = stack.pop();
            res[j] = i - j
        }
        stack.push(i)
    }
    return res
};
